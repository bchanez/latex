\begin{algorithm}[H]
  \caption{Tri par comptage}
  \begin{algorithmic}[1]
    \Statex
    \Function{TriComptage}{$tab[\;]$: tableau[1..n] d'entier, $n$: entier}
    \State {$min$ $\gets$ {$tab[1]$}}
    \State {$max$ $\gets$ {$tab[1]$}}
    \For{$i \gets 2$ to $n$} // Initialisation de $min$ et $max$
    \If{$tab[i] < min$}
    \State {$min$ $\gets$ {$tab[i]$}}
    \EndIf
    \If{$tab[i] > max$}
    \State {$max$ $\gets$ {$tab[i]$}}
    \EndIf
    \EndFor
    \State {$tabComptage$ $\gets$ {$[\;]$}}
    \For{$i \gets 1$ to $max-min+1$} // Initialisation de $tabComptage$
    \State {$tabComptage[i]$ $\gets$ {$0$}}
    \EndFor
    \For{$i \gets 1$ to $n$} // Remplit $tabComptage$
    \State {$tabComptage[tab[i]-min+1]$ $\gets$ {$tabComptage[tab[i]-min+1]+1$}}
    \EndFor
    \State {$i$ $\gets$ {$1$}}
    \For{$k \gets 1$ to $max-min+1$} // Arrange $tab$
    \For{$j \gets 1$ to $tabComptage[k]$}
    \State {$tab[i]$ $\gets$ {$k+min-1$}}
    \State {$i$ $\gets$ {$i+1$}}
    \EndFor
    \EndFor
    \State \Return {$tab$}
    \EndFunction
  \end{algorithmic}
\end{algorithm}
